001package com.genband.kandy.api.services.calls;
002
003/**
004 * This method receives the status of the call request.
005 *
006 */
007public interface KandyCallResponseListener
008{
009        /**
010         * Request successfully
011         * 
012         * @param call
013         *            the call obj
014         */
015        public void onRequestSucceeded(IKandyCall call);
016
017        /**
018         * Request failed
019         * 
020         * @param call
021         *            the call obj
022         * @param error
023         *            the error of the end call
024         */
025        public void onRequestFailed(IKandyCall call, int responseCode, String error);
026}
