001package com.genband.kandy.api.services.addressbook;
002
003
004public interface IKandyAddressBookService
005{
006
007        /**
008         * register to ongoing notification
009         * 
010         * @param listener
011         *            The callback that will be invoked
012         */
013        public void registerNotificationListener(KandyAddressBookServiceNotificationListener listener);
014
015        /**
016         * unregister from ongoing notification
017         * 
018         * @param listener
019         *            The callback that will be invoked
020         */
021        public void unregisterNotificationListener(KandyAddressBookServiceNotificationListener listener);
022        
023        /**
024         * Get device contacts.
025         * <p>
026         * 
027         * <b>Note:</b> This API requires android.permission.READ_CONTACTS
028         * 
029         * @param filters
030         *            The filter of which contacts to return. Passing null will
031         *            return all visible contacts that has phone number
032         * @param listner
033         *            The callback that will be invoked
034         * 
035         */
036        public void getDeviceContacts(KandyDeviceContactsFilter[] filters, KandyDeviceContactsListener listner);
037
038        /**
039         *  Returns the domain contacts as {@link IKandyContact} objects in an array in the response callback. If operation failed, return an error.
040         *
041         *  @param listener a callback with {@link IKandyContact} objects in an array if operation succeeded or error if operation failed.
042         */
043        public void getDomainDirectoryContacts(KandyDeviceContactsListener listener);
044        
045        /**
046         *  Returns the domain contacts as {@link IKandyContact} objects in an array in the response callback. If operation failed, return an error. Filters contact according to filterType and text in textSearch.
047         *
048         *  @param filter       the fields in which to search for the text.
049         *  @param isCaseSensitive    Boolean value indicating whether or not the search is case sensitive.
050         *  @param search_string       the text to search in the fields indicated by filterType
051         *  @param listener A callback with {@link IKandyContact} objects in an array if operation succeeded, or error if operation failed.
052         */
053        public void getFilteredDomainDirectoryContacts(KandyDomainContactFilter filter, boolean isCaseSensitive, String search_string, KandyDeviceContactsListener listener);
054
055}
