001package com.genband.kandy.api.services.common;
002
003import android.net.Uri;
004
005import com.genband.kandy.api.services.chats.IKandyTransferProgress;
006
007public abstract class KandyResponseProgressListener extends KandyBaseResponseListener
008{       
009        
010        /**
011         * Callback has been called after response succeed and pass over parameter the uri of the downloaded file on device
012         * @param fileUri URI of downloaded file
013         */
014        public abstract void onRequestSucceded(Uri fileUri);
015
016        /**
017         * This method is used to display any form of progress in the user interface
018         * while the background computation is still executing. For instance, it can
019         * be used to animate a progress bar or show logs in a text field.
020         * 
021         * @param progress The values indicating progress.
022         */
023        public abstract void onProgressUpdate(IKandyTransferProgress progress);
024
025}
