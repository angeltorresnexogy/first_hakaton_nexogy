001package com.genband.kandy.api.services.calls;
002
003import java.util.Date;
004
005import com.genband.kandy.api.services.common.KandyCameraInfo;
006import com.genband.kandy.core.services.call.api.KandyCallType;

008/**
009 * This interface is used to initiate a voice or video call.
010 * <p>
011 * When the call has been established, a call object is generated and various
012 * actions can be taken on this object. These include:
013 * <ul>
014 * <li>hangup/transfer
015 * <li>hold/unhold
016 * <li>video sharing - start/stop
017 * <li>switch camera
018 * <li>mute/unmute
019 * <li>get the call ID/state/type/callee
020 * <li>get timing info. start/end/duration/
021 * </ul>
022 * 
023 */
024public interface IKandyCall
025{
026        /**
027         * The hangup method terminates the call.
028         * 
029         * @param listener
030         *            The callback that will be invoked
031         */
032        public void hangup(KandyCallResponseListener listener);
033
034        // public void transfer(KandyCallResponseListener listener);
035        /**
036         * The hold method reports to the server that the other side of the
037         * conversation is on hold and stops sending audio.
038         * 
039         * @param listener
040         *            The callback that will be invoked
041         */
042        public void hold(KandyCallResponseListener listener);
043
044        /**
045         * The unhold method releases the other side of the conversation from being
046         * on hold and resumes sending audio.
047         * 
048         * @param listener
049         *            The callback that will be invoked
050         */
051
052        public void unhold(KandyCallResponseListener listener);
053
054        // public void changeAudioRoute(KandyCallAudioRoute route,
055        // KandyCallResponseListener listener);
056        /**
057         * The startVideoSharing method initiates the user's camera during a call.
058         * 
059         * @param listener
060         *            The callback that will be invoked
061         */
062        public void startVideoSharing(KandyCallResponseListener listener);
063
064        /**
065         * The stoptVideoSharing method disables the user's camera during a call.
066         * 
067         * @param listener
068         *            The callback that will be invoked
069         */
070        public void stopVideoSharing(KandyCallResponseListener listener);
071
072        /**
073         * The mute method refers to the subscriber side of the conversation, not
074         * the other side.
075         * 
076         * @param listener
077         *            The callback that will be invoked
078         */
079        public void mute(KandyCallResponseListener listener);
080
081        /**
082         * The unmute method refers to the subscriber side of the conversation, not
083         * the other side.
084         * 
085         * @param listener
086         *            The callback that will be invoked
087         */
088        public void unmute(KandyCallResponseListener listener);
089
090        /**
091         * Inserts Dual Tone Multi Frequency Signal to sending queue and returns the
092         * success
093         * 
094         * @param tone The character value of DTMF tone. Can only be 0,1...,9,*,#,a,b,c,d,A,B,C,D;
095         */
096        public void sendDTMF(char tone);
097
098        public String getCallId();
099
100        public KandyCallState getCallState();
101
102        public KandyCallType getCallType();
103
104        public KandyRecord getCallee();
105
106        public Date getStartTime();
107
108        public Date getEndTime();
109
110        public long getDuration();
111
112        public String getDurationString();
113
114        // public List<KandyCallParticipant> getParticipants();
115
116        // public IKandyCallParticipant getSelfParticipant();
117
118        // public List<KandyRecord> getTotalKandyRecords();
119
120        public String getVia();
121
122        public boolean isReceivingVideo();
123
124        public boolean isSendingVideo();
125
126        public boolean isIncomingCall();
127
128        // public boolean isConferenceCall();
129
130        public boolean isCallStartedWithVideo();
131
132        // public boolean isMissedCall();
133
134        // public boolean isForwarded();
135
136        public boolean isOnHold();
137
138        public boolean isOtherParticipantOnHold();
139
140        // public boolean isCallTransfer();
141
142        public KandyCallTerminationReason getTerminationReason();
143
144        // public KandyCallAudioRoute getAudioRoute();
145
146        public boolean isMute();
147
148        // public List<Object> getCameraAlternatives();
149
150        // public KandyIncomingCallAnswerType getIncomingCallAnswerType();
151
152        public void setLocalVideoView(KandyView localVideoView);
153
154        public KandyView getLocalVideoView();
155
156        public void setRemoteVideoView(KandyView remoteVideoView);
157
158        public KandyView getRemoteVideoView();
159
160        // public void setDefaultCameraForVideo(String defaultCameraForVideo);
161
162        /**
163         * @return The direction that the camera faces. It should be
164         *         {@link KandyCameraInfo#FACING_BACK} or
165         *         {@link KandyCameraInfo#FACING_BACK}.
166         */
167        public KandyCameraInfo getCameraForVideo();
168
169        /**
170         * Switch the camera faces direction
171         * @param camera The direction
172         */
173        public void switchCamera(KandyCameraInfo camera);
174
175        // public KandyVideoCodecSetting getVideoCodecSettings();
176}
