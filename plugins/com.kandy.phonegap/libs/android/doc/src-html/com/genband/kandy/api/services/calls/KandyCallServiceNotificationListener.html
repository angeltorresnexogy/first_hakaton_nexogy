006/**

008 * <p>

012 * <pre>
013 * {@code Kandy.getServices().getCallService().registerNotificationListener(listener);}

015 * 
016 * @see {@link IKandyCallService#registerNotificationListener(KandyCallServiceNotificationListener)}
017 * 
018 * 
019 */
020public interface KandyCallServiceNotificationListener

023        /**
024         * Got incoming call
025         * 
026         * @param call
027         *            the incoming call object
028         */
029        public void onIncomingCall(IKandyIncomingCall call);
030
031        /**
032         * Got missed call
033         * 
034         * @param call
035         *            the missed call object
036         * @see IKandyCall
037         */
038        public void onMissedCall(KandyMissedCallMessage call);

069        /**
